I've been working on a SASS project again this week, and decided to use Gulp to build the SASS "properly". It was a handy experience, which led me to create a simple yet effective Gulpfile.js:

const gulp = require('gulp'); const sass = require('gulp-sass')(require('sass'));

// build sass function compileSass() { return gulp.src('scss/**/compiled.scss') // Path to your SASS files .pipe(sass({ outputStyle: 'compressed', precision: 3, errLogToConsole: true, includePaths: ['node_modules/'] }).on('error', console.error)) .pipe(gulp.dest('css')); // Output directory for CSS files } gulp.task('compile-sass', compileSass); gulp.task('sass', compileSass);

// watch sass function watchSass() { gulp.watch('scss/*/.scss', gulp.series('sass')); } gulp.task('sass-watch', watchSass);

It isn't anything amazing, but I'm very pleased!
